14#[derive(Default)]
15pub struct AttributeResolver {
16    cache: Arc<RwLock<HashMap<String, Arc<Attribute>>>>,
17}
18
19impl AttributeResolver {
20    pub fn new() -> Self {
21        Self::default()
22    }
23
24    pub async fn resolve<'a, S: ReadStorage<'a>>(
25        &self,
26        storage: &'a S,
27        ident: &str,
28        tx: u64,
29    ) -> Result<Arc<Attribute>, ResolveError<S::Error>> {
30        {
31            let cache_read = self.cache.read().await;
32            if let Some(attribute) = cache_read.get(ident) {
33                return Ok(attribute.clone());
34            }
35        }
36
37        if let Some(attribute) = resolve_by_ident(storage, ident, tx)? {
38            let mut cache_write = self.cache.write().await;
39            cache_write.insert(ident.to_string(), attribute.clone());
40            return Ok(attribute);
41        }
42
43        Err(ResolveError::IdentNotFound(ident.to_string()))
44    }
45}
46
47#[derive(Debug, Error, PartialEq)]
48pub enum ResolveError<S> {
49    #[error("storage error")]
50    StorageError(#[from] S),
51    #[error("ident `{0}` not found")]
52    IdentNotFound(String),
53}
54
55fn resolve_by_ident<'a, S: ReadStorage<'a>>(
56    storage: &'a S,
57    ident: &str,
58    tx: u64,
59) -> Result<Option<Arc<Attribute>>, S::Error> {
60    // [?attribute :db/attr/ident ?ident]
61    let restricts = Restricts::new(tx)
62        .with_attribute(DB_ATTR_IDENT_ID)
63        .with_value(Value::Str(ident.to_string()));
64    if let Some(datom) = storage.find(restricts).next() {
65        return resolve_by_id(storage, datom?.entity, tx);
66    }
67    Ok(None)
68}
69
70fn resolve_by_id<'a, S: ReadStorage<'a>>(
71    storage: &'a S,
72    attribute_id: u64,
73    tx: u64,
74) -> Result<Option<Arc<Attribute>>, S::Error> {
75    let mut builder = AttributeBuilder::new(attribute_id);
76    // [?attribute _ _]
77    let restricts = Restricts::new(tx).with_entity(attribute_id);
78    for datom in storage.find(restricts) {
79        builder.consume(datom?);
80    }
81    match builder.build() {
82        Some(attribute) => Ok(Some(Arc::new(attribute))),
83        None => Ok(None),
84    }
85}

99    struct CountingStorage {
100        inner: InMemoryStorage,
101        count: Cell<usize>,
102    }
103
104    impl CountingStorage {
105        fn new() -> Self {
106            Self {
107                inner: InMemoryStorage::new(),
108                count: Cell::new(0),
109            }
110        }
111
112        fn current_count(&self) -> usize {
113            self.count.get()
114        }
115    }
116
117    impl<'a> ReadStorage<'a> for CountingStorage {
118        type Error = <InMemoryStorage as ReadStorage<'a>>::Error;
119        type Iter = <InMemoryStorage as ReadStorage<'a>>::Iter;
120
121        fn find(&'a self, restricts: Restricts) -> Self::Iter {
122            self.count.set(self.count.get() + 1);
123            self.inner.find(restricts)
124        }
125
126        fn latest_entity_id(&self) -> Result<u64, Self::Error> {
127            self.inner.latest_entity_id()
128        }
129    }
130
131    impl WriteStorage for CountingStorage {
132        type Error = <InMemoryStorage as WriteStorage>::Error;
133        fn save(&mut self, datoms: &[Datom]) -> Result<(), Self::Error> {
134            self.inner.save(datoms)
135        }
136    }
137
138    fn create_storage() -> CountingStorage {
139        let mut storage = CountingStorage::new();
140        storage.save(&default_datoms()).unwrap();
141        storage
142    }

145    async fn returns_none_when_attribute_does_not_exist() {
146        let storage = create_storage();
147        let resolver = AttributeResolver::new();
148        let result = resolver.resolve(&storage, "foo/bar", u64::MAX).await;
149        assert!(result.is_err_and(|err| matches!(err, ResolveError::IdentNotFound(_))));
150    }

152    #[tokio::test]
153    async fn resolves_existing_attribute() {
154        let mut storage = create_storage();
155
156        let resolver = AttributeResolver::new();
157        let attribute = AttributeDefinition::new("foo/bar", ValueType::U64);
158        let transaction = Transaction::new().with(attribute);
159        let tx_result = transactor::transact(&storage, &resolver, Instant(0), transaction).await;
160        assert!(tx_result.is_ok());
161        assert!(storage.save(&tx_result.unwrap().tx_data).is_ok());
162
163        let result = resolver.resolve(&storage, "foo/bar", u64::MAX).await;
164        assert!(result.is_ok());
165
166        let result = result.unwrap();
167        assert_eq!("foo/bar".to_string(), result.definition.ident);
168        assert_eq!(ValueType::U64, result.definition.value_type);
169    }

172    async fn cache_hit() {
173        let mut storage = create_storage();

175        // No calls to `CountingStorage::find` yet.

178        let resolver = AttributeResolver::new();

182        assert!(tx_result.is_ok());

184
185        let result1 = resolver.resolve(&storage, "foo/bar", u64::MAX).await;
186        assert!(result1.is_ok());
187
188        // Storage was used to resolve `foo/bar`.
189        let queries = storage.current_count();
190        assert!(queries > 0);
191
192        let result2 = resolver.resolve(&storage, "foo/bar", u64::MAX).await;
193        assert!(result2.is_ok());
194        //assert_eq!(result1, result2);
195
196        // No additional calls to storage were needed to resolve cached attribute.
